#include <stdio.h>

// Define an enumeration for days of the week
enum Weekday { Sunday, Monday, Tuesday, Wednesday, Thursday, Friday, Saturday };

int main() {
    enum Weekday today;

    today = Wednesday;

    printf("Day number of today: %d\n", today);  // Enums start from 0 by default

    if (today == Wednesday) {
        printf("It's Wednesday!\n");
    }

    return 0;
}
